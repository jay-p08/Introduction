from Crypto.Cipher import AES
import os

def encrypt_file(path, key):
    iv = os.urandom(16)
    cipher = AES.new(key, AES.MODE_CBC, iv)
    with open(path, 'rb') as f:
        data = f.read()
    encrypted = cipher.encrypt(pad(data, 16))
    return iv + encrypted
▸ 설계 의도

파일마다 다른 IV를 생성해 동일한 키로 암호화해도 암호문이 달라지도록 설계했습니다. IV는 암호문 앞에 붙여 복호화 시 분리합니다.

▸ 배운 점

CBC 모드에서 패딩이 없으면 블록 크기 불일치로 오류가 발생합니다. 암호화는 알고리즘보다 키/IV 관리 방식이 보안의 핵심임을 배웠습니다.

class BulletPool {
    std::vector<Bullet> pool;
    int maxSize;
public:
    Bullet* Acquire() {
        for (auto& b : pool)
            if (!b.active) {
                b.active = true;
                return &b;

        return nullptr; // 풀 소진

boolean checkAABB(Rect a, Rect b) {
    return a.x < b.x + b.w
        && a.x + a.w > b.x
        && a.y < b.y + b.h
        && a.y + a.h > b.y;
}
void resolveCollision(Entity e, Tile t) {
    float overlapX = Math.min(
        e.right() - t.left(),
        t.right() - e.left());
    e.x += (e.centerX() < t.centerX())
        ? -overlapX : overlapX;
}

int[][] rotate(int[][] shape) {
    int n = shape.length;
    int[][] rotated = new int[n][n];
    for (int r = 0; r < n; r++)
        for (int c = 0; c < n; c++)
            rotated[c][n-1-r] = shape[r][c];
    return rotated;
}
